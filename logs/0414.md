1. 确定基本环境：cubemx+vscode
    1. 不使用keil有卡顿
    2. cubemx生成工具链选择cmake
    3. 补充安装arm-none-eabi-gcc编译和openocd烧录
2. cmake下的基本编译流程
    1. 添加配置文件

```json
//.vscode/tasks.json
{
    "version": "2.0.0",
    "tasks": [
        {
            "label": "Build",
            "type": "shell",
            "command": "cmake --build build/Debug",
            "group": { "kind": "build", "isDefault": true },
            "problemMatcher": "$gcc"
        }
    ]
}
//.vscode/launch.json
{
    "version": "0.2.0",
    "configurations": [
        {
            "name": "Debug STM32G4",
            "cwd": "${workspaceRoot}",
            "executable": "build/Debug/${workspaceFolderBasename}.elf",
            "request": "launch",
            "type": "cortex-debug",
            "servertype": "openocd",
            "preLaunchTask": "Build",
            "configFiles": [
                "interface/stlink.cfg",
                "target/stm32g4x.cfg"   // G4 系列专用配置
            ],
            "svdFile": "${workspaceRoot}/STM32G474xx.svd",  // 可选，寄存器查看
            "showDevDebugOutput": "none"
        }
    ]
}
//.vscode/settings.json
{
    "cmake.configureOnOpen": true,
    "cmake.generator": "Ninja",
    "C_Cpp.default.configurationProvider": "ms-vscode.cmake-tools"
}
```
​        2.编译烧录

```powershell
cmake --preset Debug#配置生成构建系统
cmake --build build/Debug
openocd -f interface/stlink.cfg -f target/stm43g4.cfg -c "program build/Debug/g431test_cmake.elf verify reset exit"
```

3. 添加组件

```json
# Add sources to executable
target_sources(${CMAKE_PROJECT_NAME} PRIVATE
    # Add user sources here
    App/Src/bms_main.c
)

# Add include paths
target_include_directories(${CMAKE_PROJECT_NAME} PRIVATE
    # Add user defined include paths
    App/Inc
)
```
